f=[[0,0,0,0,0],[12,5,3,4,1],[0,0,5,2,1],[9,0,4,0,1]]


print(f)

def sifirBul(matris):
    sonuc= []
    for i in range(0,len(matris)):
        for j in range(0,len(matris[i])):
            if j+1 >= len (matris[i]):
                break
            else:
                if matris[i][j] == matris[i][j+1]:
                    sonuc.append(i)
                    break
    return sonuc


print(sifirBul(f)," satirlarÄ± bulundu")
